#include <bits/stdc++.h>
using namespace std;
 int main() {
 int t,j;
 cout<<"Enter the no of testcases:";
 cin >> t;
 while(t--)
 {
 int n;
 cout<<"Enter N:";
 cin >> n;
 int a[n];
 cout<<"Enter elements:";
 for(int i=0 ; i<n ;i++)
 {
 cin >> a[i];
 }
 int w=0 ,h=0;
 int flag=0;
 for(int i=0 ; i<n ;i++)
 {
 w=0;
 h=0;
 
 for( j=0 ; j<n ;j++)
 {
 if(a[j]> a[i])
 {
 w++;
 }
 if(a[j] <= a[i])
 {
 h++;
 }
 
 }
 if(h>w)
 {
 flag++;
 }
 }
 
 
 cout << flag << "\n";
 }
 // your code goes here
 return 0;
}
