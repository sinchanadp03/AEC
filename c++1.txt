#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cout<<"Enter the number of testcases:";
    cin>>t;
    while(t--)
    {
        int x,y;
        cout<<"Enter the value of x & y:";
        cin>>x>>y;
        if(x<y)
        {
            cout<<x<<endl;
        }
        else
        {
            cout<<(x/y)+(x%y)<<endl;
        }
    }
    return 0;
}
